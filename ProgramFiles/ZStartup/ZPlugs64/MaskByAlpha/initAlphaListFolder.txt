
/* Create or clear subfolder to sotre data
 *
 *
 */
[RoutineDef, initAlphaListFolder
,
	[VarSet, folder_path, [FileNameResolvePath, $ALPHA_LIST_FOLDER]]

	[VarSet,folderExists, [FileExecute, $ZFILE_UTILS_DLL, "FolderExists", #folder_path]]

	[If, ! folderExists
	, // THEN
		[VarSet, err, [FileExecute, $ZFILE_UTILS_DLL, "MakeFolder", #folder_path]]	// create the folder

	, // ELSE
		[VarSet, err, [FileExecute, $ZFILE_UTILS_DLL, "EmptyFolder", #folder_path]] // delete files in folder
	]
] // END deleteFolder()
